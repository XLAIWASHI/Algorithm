## 带权
**带权** 的实现思路很简单，就是按 **秩** ，永远将深度低的放在深度高的下面，避免出现树深度过高的情况

## 带权代码
```cpp
//UF类
#include <vector>

class UF
{
public:
    UF(int n);
    ~UF();
    void unite(int x, int y);
    bool isConnected(int x, int y);
private:
    int root(int i);
    std::vector<int> id;
    std::vector<int> sz;
};
```

```cpp
//UF类实现
#include <iostream>
#include "UnionFind.h"

UF::UF(int n)
{
    id.resize(n);
    for(int i = 0; i < n; i++)
    {
        id[i] = i;
    }
    sz.resize(n, 1);
}

UF::~UF()
{
}

void UF::unite(int x, int y)
{
    int i = root(x);
    int j = root(y);
    if(i == j) //有sz数组，所以如果已经连接，就需要返回，否则sz数组会乱
    {
        return;
    }
    if(sz[i] < sz[j])
    {
        id[i] = j;
        sz[j] += sz[i];
    }
    else
    {
        id[j] = i;
        sz[i] += sz[j];
    }
}

bool UF::isConnected(int x, int y)
{
    return root(x) == root(y);
}

int UF::root(int i)
{
    while(i != id[i])
    {
        i = id[i];
    }
    return i;
}
```
